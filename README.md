public class Sala {
    
    private Alumno alumnos[];
    private int df;
    private int dl;
    
    public Sala(int m){
        this.df = m;
        this.dl = 0;
        alumnos = new Alumno[df];
    }
    
    public void AgregarAlumno(Alumno a){

    }
    
    public int getCant(){
        return dl;

        GeneradorAleatorio.iniciar();

    public String toString(int t){

                aux = aux + alumnos[i] + " ";
        }
        return aux;
    }
